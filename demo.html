Fenix Database function

Code sample

Include library:

include_once('ini_db.php');

Next step is to create db object

$baseApi = new Fenix_baseApi();

Select table and return value

  • $arr_base = array (
    • 'table' => 'test_base', // name of table
    • 'select' => array( // select array of table row witch wont to select
      • 'id',
      • 'text'
    • ),
    • 'where' => 'text = "djuro" ' // where condition see manuel for database
    • 'order' => 'asc' // how to sory returned value
  • );

call function in object

$value = $baseApi -> fx_select( $arr_base );

Insert value into table

  • $arr_base_insert = array(
    • 'table' => 'new_table4',
    • 'value' => array(
      • 'FirstName' => 'dexter', // FirstName is tablerow name where wont to insert 'dexter' text
      • 'LastName' => 'more test',
      • 'Age' => '2'
    • )
  • );

Call function in object

$value2 = $baseApi -> fx_insert( $arr_base_insert );

Delete data from table

  • $arr_base_delete = array(
    • 'table' => 'test_base', // name of table
    • 'where' => ' id = 8 ' // condition where like mysql where
  • );

$val3 = $baseApi -> fx_delete($arr_base_delete); // call function into object

Create table into database

  • $arr_base_create_table = array(
    • 'table' => 'new_table4', // name of new table
    • 'tableType' => 'innodb', // type of table
    • 'value' => array(
      • 'id' => 'int NOT NULL AUTO_INCREMENT', // if id not defined script automatic add this field
      • 'FirstName' => 'varchar(15)', // first name is name of row witch wont to sreate second is type and length
      • 'LastName' => 'varchar(15)',
      • 'Age' => 'int'
    • ),
    • 'primary' => 'id' // if primary key not defined id automatic becomes primary key
  • );

$value3 = $baseApi -> fx_createTable($arr_base_create_table);

List of all table in db

$list = $baseApi -> fx_tableList();

Update value

  • $arr_base_update = array(
    • 'table' => 'new_table', // table witch want to update
    • 'value' => array(
      • 'FirstName' => 'sasa', FirstName is name of table row sasa is a value for update
      • 'LastName' => 'des'
    • ),
    • 'where' => "id = 1 or age = 22" // condition witch field to update
  • );

$update = $baseApi -> fx_update( $arr_base_update );

Drop table

  • $arr_base_drop = array(
    • 'table' => 'new_table1' // table for drop
  • );

$drop = $baseApi -> fx_drop( $arr_base_drop );
